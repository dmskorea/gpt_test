select 

dmag01004.기준년월,
dmag01004.기준년도,
dmag01004.기준월,
dmag01004.계약id,
dmag01004.증권번호,
dmag01004.자동차기본담보코드,
dmag01004.보험기간시작일자,
dmag01004.보험기간종료일자,
tt02053.자동차적용보험료


from (
select distinct
t1.기준년월 ,
기준년월.기준년도,
기준년월.기준월,
t1.계약id,
t1.증권번호,
t1.자동차기본담보코드,
t1.보험기간시작일자,
t1.보험기간종료일자
from 자동차원계약담보명세_월 t1, 기준년월
where 1=1
and t1.기준년월 = '202201'
and t1.자동차기본담보코드 in ('01','02','03','06')
and t1.상품코드 in ('08','19')
and t1.기준년월 = 기준년월.기준년월
) dmag01004
left outer join (
select
t1.계약id,
t1.기본담보코드,
t1.유효시작일자,
sum(t1.자동차적용보험료)
from 계약자동차보험료산출내역 t1
where 1=1
and t1.자동차기본담보코드 in ('01','02','03','06')
group by t1.계약id, t1.기본담보코드, t1.유효시작일자

) tt02053

on (((damg01004.계약id=tt02053.계약id) and (damg01004.자동차기본담보코드=tt02053.자동차기본담보코드)) and (damg01004.보험기간시작일자=tt02053.유효시작일자))
